#include <BluetoothSerial.h>
#include <WiFi.h>

BluetoothSerial BT;

void setup() {
  Serial.begin(115200);
  BT.begin("WiFiConfigDevice"); // 设备名称
  Serial.println("[BT] Bluetooth initialized");
}

void loop() {
  if (BT.hasData()) {
    String data = BT.readString();
    Serial.print("[RX] ");
    Serial.println(data);
